  <ul>
    <li><strong>Frontend:</strong> React.js with Vite</li>
    <li><strong>Backend:</strong> Node.js with Express.js</li>
    <li><strong>Database:</strong> MongoDB</li>
    <li><strong>Authentication:</strong>
      <ul>

      </ul>
    </li>
